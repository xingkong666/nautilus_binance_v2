策略参数网格扫描

options:
  -h, --help            show this help message and exit
  --strategy {all,ema,rsi,micro_scalp,vegas_tunnel,turtle,ema_pullback_atr}
  --interval {1m,5m,15m,1h,4h}
                        K 线间隔（默认：15m）
  --symbols SYMBOLS [SYMBOLS ...]
                        扫描标的列表（默认：BTCUSDT ETHUSDT）
  --is-start IS_START   In-sample 起始日期
  --is-end IS_END       In-sample 结束日期
  --oos-start OOS_START
                        OOS 起始日期
  --oos-end OOS_END     OOS 结束日期
  --oos-top-k OOS_TOP_K
                        每个 symbol 的 OOS 候选数量
  --no-oos              跳过 OOS 验证
  --workers WORKERS     并行进程数
  --output-dir OUTPUT_DIR
                        结果输出目录
  --top-n TOP_N         排行表展示前 N 名
  --save                保存最优结果到
                        experiments/param_sweep/<strategy>_<symbol>_best.json
